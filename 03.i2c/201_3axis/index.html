# coding: utf-8
import FaBo3Axis_ADXL345
import time
import sys

adxl345 = FaBo3Axis_ADXL345.ADXL345()

try:
    while True:
        axes = adxl345.read()
        sys.stdout.write("\rx = %f, y = %f, z = %f" % (axes['x'],  axes['y'], axes['z']))
        sys.stdout.flush()
        time.sleep(0.1)
except KeyboardInterrupt:
    sys.exit()
